Create `lib/rfp/sources.ts` as the single source of truth for RFP API endpoints. Phase 5 will import from here. Define one record per source with: `name`, `requiresAuth`, `authMethod` (`query_param | header | none`), `baseUrl`, `searchPath`, and a typed `pingRequest()` helper that returns a Request object suitable for a smoke test (NOT a full search — just the lightest call that returns 200 when auth is good and 401/403 when it's not).

```typescript
import { env } from '@/lib/env';

export type RfpSourceName = 'sam_gov' | 'grants_gov' | 'simpler_grants' | 'sbir_gov';

export interface RfpSource {
  name: RfpSourceName;
  displayName: string;
  requiresAuth: boolean;
  pingRequest: () => Request;
  // Returns true if status indicates auth-OK (200 or known-good 4xx that still proves auth worked, e.g. 400 missing-param after auth pass).
  isAuthOk: (status: number) => boolean;
}

export const RFP_SOURCES: Record<RfpSourceName, RfpSource> = {
  sam_gov: {
    name: 'sam_gov',
    displayName: 'SAM.gov',
    requiresAuth: true,
    pingRequest: () => {
      const key = env.SAM_GOV_API_KEY;
      if (!key) throw new Error('SAM_GOV_API_KEY not set');
      // Search v2 with a tiny date window — returns 200 when key is valid, 401 when invalid, regardless of result count.
      const url = new URL(`${env.SAM_GOV_BASE_URL}/search`);
      url.searchParams.set('api_key', key);
      url.searchParams.set('postedFrom', '01/01/2026');
      url.searchParams.set('postedTo', '01/02/2026');
      url.searchParams.set('limit', '1');
      return new Request(url.toString(), { method: 'GET' });
    },
    isAuthOk: (s) => s === 200,
  },
  grants_gov: {
    name: 'grants_gov',
    displayName: 'Grants.gov',
    requiresAuth: false,
    pingRequest: () => {
      // Grants.gov search2 is POST JSON with empty filters → returns 200 with metadata.
      return new Request(`${env.GRANTS_GOV_BASE_URL}/search2`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ rows: 1, keyword: 'test' }),
      });
    },
    isAuthOk: (s) => s === 200,
  },
  simpler_grants: {
    name: 'simpler_grants',
    displayName: 'Simpler.Grants.gov',
    requiresAuth: true,
    pingRequest: () => {
      const key = env.SIMPLER_GRANTS_API_KEY;
      if (!key) throw new Error('SIMPLER_GRANTS_API_KEY not set');
      return new Request(`${env.SIMPLER_GRANTS_BASE_URL}/opportunities/search`, {
        method: 'POST',
        headers: { 'X-Auth': key, 'Content-Type': 'application/json' },
        body: JSON.stringify({ pagination: { page_size: 1, page_offset: 1, sort_order: [{ order_by: 'opportunity_id', sort_direction: 'ascending' }] } }),
      });
    },
    isAuthOk: (s) => s === 200,
  },
  sbir_gov: {
    name: 'sbir_gov',
    displayName: 'SBIR.gov',
    requiresAuth: false,
    pingRequest: () => {
      return new Request(`${env.SBIR_GOV_BASE_URL}/solicitations.json?rows=1`, { method: 'GET' });
    },
    isAuthOk: (s) => s === 200,
  },
};
```

  <name>Task 3: Create scripts/smoke-test-rfp-apis.ts and wire npm script</name>
  <files>scripts/smoke-test-rfp-apis.ts, package.json</files>
  <action>
Create `scripts/smoke-test-rfp-apis.ts` — a CLI script that runs each source's ping request, prints a per-source result line, and exits 0 only if every source either passes auth or is configured-as-skipped.

Behavior:
- For sources where the API key is present in env: run the ping. Pass if `isAuthOk(status)` is true. Fail with `[FAIL]` if status is 401 or 403 ("auth bad"). Fail with `[ERROR]` for network/parse exceptions.
- For sources where the API key is in `requiresAuth=true` but env var is missing: print `[SKIP] {source}: env var not set (re-register pending)`, do NOT fail the run, but exit code 1 if ALL auth-required sources are skipped (we need at least one federal source live).
- Treat `requiresAuth=false` sources strictly — they should always pass.

Implementation:

```typescript
#!/usr/bin/env tsx
import 'dotenv/config';
import { RFP_SOURCES, type RfpSourceName } from '../lib/rfp/sources';
import { env } from '../lib/env';

type Result = { source: RfpSourceName; status: 'pass' | 'fail' | 'skip' | 'error'; httpStatus?: number; message?: string };

async function ping(name: RfpSourceName): Promise<Result> {
  const src = RFP_SOURCES[name];
  // Skip if auth-required and key missing
  if (name === 'sam_gov' && !env.SAM_GOV_API_KEY) return { source: name, status: 'skip', message: 'SAM_GOV_API_KEY not set (re-registration pending)' };
  if (name === 'simpler_grants' && !env.SIMPLER_GRANTS_API_KEY) return { source: name, status: 'skip', message: 'SIMPLER_GRANTS_API_KEY not set' };
  try {
    const req = src.pingRequest();
    const res = await fetch(req);
    if (src.isAuthOk(res.status)) return { source: name, status: 'pass', httpStatus: res.status };
    return { source: name, status: 'fail', httpStatus: res.status, message: await res.text().then(t => t.slice(0, 200)).catch(() => '') };
  } catch (e) {
    return { source: name, status: 'error', message: e instanceof Error ? e.message : String(e) };
  }
}

async function main() {
  console.log('RFP API Smoke Test —', new Date().toISOString());
  console.log('---');
  const sources: RfpSourceName[] = ['sam_gov', 'grants_gov', 'simpler_grants', 'sbir_gov'];
  const results = await Promise.all(sources.map(ping));
  for (const r of results) {
    const tag = r.status === 'pass' ? '[PASS]' : r.status === 'skip' ? '[SKIP]' : r.status === 'fail' ? '[FAIL]' : '[ERROR]';
    console.log(`${tag} ${r.source} ${r.httpStatus ?? ''} ${r.message ?? ''}`);
  }
  console.log('---');
  const noAuthSources = results.filter(r => RFP_SOURCES[r.source].requiresAuth === false);
  const authSources = results.filter(r => RFP_SOURCES[r.source].requiresAuth === true);
  const noAuthAllPass = noAuthSources.every(r => r.status === 'pass');
  const atLeastOneAuthPass = authSources.some(r => r.status === 'pass');
  const anyAuthFail = authSources.some(r => r.status === 'fail' || r.status === 'error');

  if (!noAuthAllPass) {
    console.error('FAIL: Public-API sources (Grants.gov, SBIR.gov) must always pass — they have no auth.');
    process.exit(2);
  }
  if (anyAuthFail) {
    console.error('FAIL: An auth-required source returned a non-401 failure — investigate.');
    process.exit(3);
  }
  if (!atLeastOneAuthPass && authSources.every(r => r.status === 'skip')) {
    console.warn('WARN: All auth-required sources are skipped (keys not yet provisioned). Re-run after keys land.');
    process.exit(1);
  }
  console.log('OK: smoke test passed.');
  process.exit(0);
}

main();
```
