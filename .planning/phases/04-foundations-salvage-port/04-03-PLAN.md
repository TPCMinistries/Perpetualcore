Create `lib/rfp/orgs.ts` with two server-side functions:

```typescript
import { createClient, createAdminClient } from '@/lib/supabase/server';

export type OrgType = 'nonprofit' | 'forprofit' | 'dual';
export type OrgRole = 'owner' | 'writer' | 'reviewer' | 'viewer';

export interface CreateOrgInput {
  name: string;
  type: OrgType;
  naics: string[];
}

export async function createOrgWithOwner(input: CreateOrgInput, userId: string) {
  // Use admin client because we need atomic insert into two tables.
  // RLS on rfp_orgs.INSERT is `WITH CHECK (true)` (per Plan 04-01) but rfp_user_orgs.INSERT
  // requires the user be inserting their own row OR be an owner of the org.
  // Since the org doesn't exist yet, we use the admin client to bypass for THIS specific
  // bootstrap insert. Justified per CLAUDE.md: "Background/server operations: ALWAYS use createAdminClient()".
  const admin = createAdminClient();
  const { data: org, error: orgErr } = await admin
    .from('rfp_orgs')
    .insert({ name: input.name, type: input.type, naics: input.naics })
    .select()
    .single();
  if (orgErr || !org) throw new Error(`org_create_failed: ${orgErr?.message}`);

  const { error: memErr } = await admin
    .from('rfp_user_orgs')
    .insert({ user_id: userId, org_id: org.id, role: 'owner' });
  if (memErr) {
    // Compensate: delete the org we just created.
    await admin.from('rfp_orgs').delete().eq('id', org.id);
    throw new Error(`org_membership_failed: ${memErr.message}`);
  }
  return org;
}

export async function getOrgForUser(orgId: string) {
  // Anon-keyed client; RLS does the filtering. Returns null if no access.
  const supabase = await createClient();
  const { data } = await supabase.from('rfp_orgs').select('*').eq('id', orgId).maybeSingle();
  return data;
}

export async function listUserOrgs() {
  const supabase = await createClient();
  const { data } = await supabase
    .from('rfp_user_orgs')
    .select('role, rfp_orgs(id, name, type, naics)')
    .order('created_at', { ascending: false });
  return data ?? [];
}
```

Then create `app/api/orgs/route.ts`:

```typescript
import { NextResponse } from 'next/server';
import { z } from 'zod';
import { createClient } from '@/lib/supabase/server';
import { createOrgWithOwner, listUserOrgs } from '@/lib/rfp/orgs';

const CreateOrgSchema = z.object({
  name: z.string().min(2).max(120),
  type: z.enum(['nonprofit', 'forprofit', 'dual']),
  naics: z.array(z.string().regex(/^\d{2,6}$/)).max(20).default([]),
});

export async function POST(req: Request) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: 'unauthorized' }, { status: 401 });

  let body;
  try { body = CreateOrgSchema.parse(await req.json()); }
  catch (e) { return NextResponse.json({ error: 'invalid_body', detail: String(e) }, { status: 400 }); }

  try {
    const org = await createOrgWithOwner(body, user.id);
    return NextResponse.json({ org }, { status: 201 });
  } catch (e) {
    return NextResponse.json({ error: 'create_failed', detail: e instanceof Error ? e.message : String(e) }, { status: 500 });
  }
}

export async function GET() {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: 'unauthorized' }, { status: 401 });
  const orgs = await listUserOrgs();
  return NextResponse.json({ orgs });
}
```

The CLAUDE.md rule says "Background/server operations: ALWAYS use createAdminClient()". This is exactly that case — a server-side privileged action that the user authenticated to perform.

If `lib/supabase/server.ts` doesn't export `createAdminClient` under that exact name, use whatever name it actually exports (we've already confirmed `createAdminClient` is exported on line 53). Do NOT add `'use server'` — these are route handlers and helpers, not Server Actions.

Create `components/rfp/CreateOrgForm.tsx`:

```tsx
'use client';

import { useState } from 'react';
import { useRouter } from 'next/navigation';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';

type OrgType = 'nonprofit' | 'forprofit' | 'dual';

export function CreateOrgForm() {
  const router = useRouter();
  const [name, setName] = useState('');
  const [type, setType] = useState<OrgType>('nonprofit');
  const [naicsInput, setNaicsInput] = useState('');
  const [naics, setNaics] = useState<string[]>([]);
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const addNaics = () => {
    const code = naicsInput.trim();
    if (/^\d{2,6}$/.test(code) && !naics.includes(code)) {
      setNaics([...naics, code]);
      setNaicsInput('');
    }
  };

  const onSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setSubmitting(true);
    setError(null);
    const res = await fetch('/api/orgs', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, type, naics }),
    });
    const data = await res.json();
    if (!res.ok) {
      setError(data.error ?? 'Something went wrong');
      setSubmitting(false);
      return;
    }
    router.push(`/org/${data.org.id}`);
  };

  return (
    <form onSubmit={onSubmit} className="space-y-6">
      <div>
        <Label htmlFor="name">Organization name</Label>
        <Input id="name" value={name} onChange={(e) => setName(e.target.value)} required minLength={2} maxLength={120} />
      </div>
      <div>
        <Label htmlFor="type">Type</Label>
        <Select value={type} onValueChange={(v) => setType(v as OrgType)}>
          <SelectTrigger id="type"><SelectValue /></SelectTrigger>
          <SelectContent>
            <SelectItem value="nonprofit">Nonprofit (501c3)</SelectItem>
            <SelectItem value="forprofit">For-profit</SelectItem>
            <SelectItem value="dual">Dual (operates both)</SelectItem>
          </SelectContent>
        </Select>
      </div>
      <div>
        <Label htmlFor="naics">NAICS codes</Label>
        <div className="flex gap-2">
          <Input id="naics" value={naicsInput} onChange={(e) => setNaicsInput(e.target.value)} placeholder="e.g. 541512" />
          <Button type="button" variant="outline" onClick={addNaics}>Add</Button>
        </div>
        <div className="mt-2 flex flex-wrap gap-1">
          {naics.map((code) => (
            <span key={code} className="rounded-full bg-muted px-3 py-1 text-sm">
              {code}
              <button type="button" className="ml-2 text-muted-foreground" onClick={() => setNaics(naics.filter((c) => c !== code))}>×</button>
            </span>
          ))}
        </div>
      </div>
      {error && <p className="text-sm text-destructive">{error}</p>}
      <Button type="submit" disabled={submitting || !name || naics.length === 0}>
        {submitting ? 'Creating…' : 'Create organization'}
      </Button>
    </form>
  );
}
```

Use `@/components/ui/{button,input,label,select}` — these are existing shadcn components in the project (verified by `package.json` having `@radix-ui/react-select` and shadcn-pattern conventions).
