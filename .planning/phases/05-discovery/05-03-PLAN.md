```ts
export const WEIGHTS = {
  naics: 0.30,
  keyword: 0.25,
  geo: 0.20,
  dollar_band: 0.15,
  past_funder: 0.10,
} as const;

export const TIER_THRESHOLDS = {
  strong: 90,    // ≥90 → "Strong fit", emerald
  good: 70,      // 70-89 → "Good fit", zinc-300
  marginal: 50,  // 50-69 → "Marginal", zinc-500
  // <50 → "Weak"
} as const;

export type FitTier = 'Strong fit' | 'Good fit' | 'Marginal' | 'Weak';

export function tierFor(score: number): FitTier {
  if (score >= TIER_THRESHOLDS.strong) return 'Strong fit';
  if (score >= TIER_THRESHOLDS.good) return 'Good fit';
  if (score >= TIER_THRESHOLDS.marginal) return 'Marginal';
  return 'Weak';
}
```

These constants are FROZEN — user-tunable weights are explicitly deferred to Phase 7/10 per CONTEXT.md.

**1c. `lib/rfp/scoring/score.ts`** — pure deterministic function (no DB calls, no AI):

```ts
export interface OpportunityForScoring {
  id: string;
  source: string;
  title: string;
  agency: string | null;
  amount_min: number | null;
  amount_max: number | null;
  deadline: string | null;
  brief: string | null;
  keywords: string[];
  geo: string | null;
}

export interface CaptureProfileForScoring {
  naics: string[];                          // from rfp_orgs.naics
  capacity_keywords: string[];              // from rfp_capture_profiles.profile_json -> capacity terms
  geo_focus: string[];                      // home geographies, e.g. ['NY','NYC','US']
  typical_award_band: { min: number; max: number } | null;  // from past wins
  past_funders: string[];                   // agency names from past wins
}

export interface ScoreBreakdown {
  naics: { score: number; matched: string[] };
  keyword: { score: number; matched: string[] };
  geo: { score: number; opp_geo: string | null; org_geos: string[] };
  dollar_band: { score: number; opp_amount: number | null; band_min: number | null; band_max: number | null };
  past_funder: { score: number; matched: string | null };
  fit_score: number;     // 0-100, weighted sum * 100
  chips: string[];       // 4 short chips for UI
  profile_pending: boolean;  // true when CaptureProfileForScoring is empty/default
}

export function scoreOpportunity(
  opp: OpportunityForScoring,
  profile: CaptureProfileForScoring | null
): ScoreBreakdown
```

**Scoring component rules** (each component is in [0, 1], multiplied by weight, summed, ×100):

  1. **NAICS overlap (0.30)**: `intersection(opp.keywords ∩ profile.naics).length / max(profile.naics.length, 1)`. The ingestion in 05-01 dropped opp NAICS into `keywords` for SAM.gov; for non-SAM sources this naturally returns 0, which is fine.
     - Chip when matched: `"NAICS match"` (or `"NAICS: 541611"` if you prefer specificity — pick one).

  2. **Keyword alignment (0.25)**: `intersection(opp.keywords ∪ tokens(opp.title) ∪ tokens(opp.brief), profile.capacity_keywords).length / max(profile.capacity_keywords.length, 1)`. Tokenize via lowercase + split on non-word, then dedupe. Cap at 1.0.
     - Chip when score > 0.5: `"Capacity OK"` or `"Strong keyword match"`.

  3. **Geo match (0.20)**: 1.0 if `opp.geo` ∈ `profile.geo_focus`; 0.5 if opp.geo == 'US' AND any geo_focus entry is a US sub-region; else 0.0.
     - Chip when matched: `"Geo match: ${opp.geo}"`.

  4. **Dollar-band (0.15)**: Compute `opp_amount = opp.amount_max ?? opp.amount_min`. If null OR profile.typical_award_band null → 0.0. If `band.min ≤ opp_amount ≤ band.max * 1.5`, score = 1.0. If within ±2× of band, score = 0.5. Else 0.0.
     - Chip when 1.0: `"In award band"`.

  5. **Past funder (0.10)**: 1.0 if `opp.agency` (case-insensitive substring match) appears in `profile.past_funders`. Else 0.0.
     - Chip when 1.0: `"${matched_count} prior wins"` (count is opportunity-level — set to "Prior funder match" if simpler).

**Bonus chip** (always added for context):
  - Deadline workability: if deadline > 4 weeks out → `"Workable deadline"`; if 1-4 weeks → `"Tight deadline"`; if <1 week → `"Imminent deadline"`. If null, omit.

Limit final chips to **exactly 4** (CONTEXT.md says "NAICS match · Capacity OK · 3 prior wins · Workable deadline"). Pick the 4 with the strongest signal: prefer matched-positive chips over zero-signal ones; always include the deadline chip if available.

**Profile-pending fallback**: when `profile === null` OR `profile.naics.length === 0 && profile.capacity_keywords.length === 0`, return:
  - fit_score = 50 (neutral, sits in "Marginal" tier so the user can still see the row but knows nothing is dialed in)
  - chips: build a chip array of EXACTLY length 4 (FeedRow renderer assumes 4 — must match non-pending path):
    1. `'Profile pending'` — always.
    2. `'Geo: ' + (opp.geo ?? 'unknown')` — always (use 'Geo: unknown' if opp.geo is null).
    3. Deadline chip if deadline is non-null (`'Workable deadline'` / `'Tight deadline'` / `'Imminent deadline'` per the rules above). When `opp.deadline === null`, substitute a **source-tier fallback chip**: derive from `opp.source` →
       - `'sam_gov' | 'grants_gov' | 'simpler_grants' | 'sbir'` → `'Federal'`
       - `'ny_state'` → `'NY State'`
       - `'nyc_dycd' | 'nyc_hra' | 'nyc_doe'` → `'NYC'`
       - `'foundation_url'` → `'Foundation'`
       - default → `'Other source'`
    4. Source-tier chip (same mapping as above) — always added as the 4th slot for parity with non-pending rows. If slot 3 is already a source-tier chip (because deadline was null), use a literal `'Imported'` for foundation_url or repeat the tier — implementer's judgment to avoid duplication. Whatever the choice, **the array MUST be length 4 every time scoreOpportunity returns** so FeedRow.tsx (05-04) can render `chips[0..3]` without index guards.
  - profile_pending = true
  - Include only the geo and deadline components in score_breakdown; others get score=0 with note 'profile_pending'.

**Non-pending chip count parity:** the non-pending path also MUST always emit exactly 4 chips. If after picking the strongest-signal positive chips the array length is < 4, pad with the same source-tier fallback (`'Federal'` / `'NY State'` / `'NYC'` / `'Foundation'` / `'Other source'`) until length === 4. This guarantees FeedRow renderer never has to handle variable-length arrays.

The function is **pure**: no DB, no AI, no fetch. Unit-testable in isolation.
